```typescript
async function migrateFarmData(vault: VaultManager, baseProfiles: Array<{id: string; name: string; [k: string]: any}>) {
  // Pour chaque profil, si farm-{id}.md n'existe pas mais famille.md contient des champs farm...
  const familleContent = await vault.readFile(FAMILLE_FILE);
  // Parser famille.md avec l'ANCIEN parseur (avant suppression des champs farm)
  // ATTENTION : parseFamille ne lit plus les champs farm (Task 1).
  // Donc il faut parser manuellement les lignes de famille.md pour extraire les champs farm.
  // Pattern : lire les lignes, trouver ### {profileId}, collecter les key:value farm_*/mascot_*/companion/tree_species
  for (const profile of baseProfiles) {
    const farmPath = farmFile(profile.id);
    const exists = await vault.readFile(farmPath).then(() => true).catch(() => false);
    if (exists) continue; // deja migre

    // Extraire les champs farm/mascot/companion de la section ### {profile.id} dans famille.md
    const lines = familleContent.split('\n');
    const farmFields: Record<string, string> = {};
    let inSection = false;
    for (const line of lines) {
      if (line.startsWith('### ')) {
        if (inSection) break;
        if (line.replace('### ', '').trim() === profile.id) inSection = true;
      } else if (inSection && line.includes(': ')) {
        const colonIdx = line.indexOf(': ');
        const key = line.slice(0, colonIdx).trim();
        const val = line.slice(colonIdx + 2).trim();
        const farmKeys = ['tree_species','mascot_decorations','mascot_inhabitants','mascot_placements',
          'farm_crops','farm_buildings','farm_inventory','farm_harvest_inventory',
          'farm_crafted_items','farm_tech','farm_rare_seeds','wear_events','companion'];
        if (farmKeys.includes(key) && val) farmFields[key] = val;
      }
    }

    // Si aucun champ farm trouve, creer un fichier vide quand meme (pour ne pas re-migrer)
    // Parser les champs extraits avec les parsers existants pour construire FarmProfileData
    // Serialiser avec serializeFarmProfile et ecrire farm-{id}.md
    await vault.writeFile(farmPath, serializeFarmProfile(profile.name, parsedFarmData));
  }
}
```

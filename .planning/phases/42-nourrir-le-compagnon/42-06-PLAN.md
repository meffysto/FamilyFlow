    ```typescript
    /**
     * FeedPicker.tsx — Sheet picker crops pour nourrir le compagnon (Phase 42 D-02)
     *
     * Liste toutes les combinaisons (cropId, grade) ayant qty > 0 dans l'inventaire.
     * Affiche badge affinité (préféré ❤️ / détesté 😖) selon l'espèce compagnon.
     * Détestés tappables mais grisés (D-03) — le joueur peut gâcher un crop.
     */

    import React, { useMemo, useCallback } from 'react';
    import { View, Text, Pressable, ScrollView, Modal, StyleSheet } from 'react-native';
    import { SafeAreaView } from 'react-native-safe-area-context';
    import * as Haptics from 'expo-haptics';
    import { useTranslation } from 'react-i18next';
    import { useThemeColors } from '../../contexts/ThemeContext';
    import { ModalHeader } from '../ui';
    import { FontSize, FontWeight } from '../../constants/typography';
    import { Spacing, Radius } from '../../constants/spacing';
    import { CROP_CATALOG, type HarvestInventory } from '../../lib/mascot/types';
    import {
      getAffinity,
      type HarvestGrade,
      type CompanionSpecies,
      type CropAffinity,
    } from '../../lib/mascot/companion-types';

    const GRADES_ORDER: HarvestGrade[] = ['perfect', 'excellent', 'good', 'ordinary'];

    const GRADE_EMOJI: Record<HarvestGrade, string> = {
      ordinary:  '🟫',
      good:      '🟢',
      excellent: '🔵',
      perfect:   '⭐',
    };

    const GRADE_LABEL: Record<HarvestGrade, string> = {
      ordinary:  'Ordinaire',
      good:      'Bon',
      excellent: 'Excellent',
      perfect:   'Parfait',
    };

    const AFFINITY_RANK: Record<CropAffinity, number> = {
      preferred: 0,
      neutral:   1,
      hated:     2,
    };

    interface FeedPickerProps {
      visible: boolean;
      onClose: () => void;
      inventory: HarvestInventory;
      companionSpecies: CompanionSpecies;
      onPick: (cropId: string, grade: HarvestGrade) => void;
    }

    interface Row {
      cropId: string;
      emoji: string;
      labelKey: string;
      grade: HarvestGrade;
      qty: number;
      affinity: CropAffinity;
    }

    export function FeedPicker({
      visible,
      onClose,
      inventory,
      companionSpecies,
      onPick,
    }: FeedPickerProps) {
      const { t } = useTranslation();
      const { colors, primary } = useThemeColors();

      const rows: Row[] = useMemo(() => {
        const out: Row[] = [];
        for (const [cropId, entry] of Object.entries(inventory || {})) {
          const def = CROP_CATALOG.find(c => c.id === cropId);
          if (!def) continue;
          // entry shape Phase B : Partial<Record<HarvestGrade, number>>
          // Legacy shape possible : number direct (backward compat) — on traite comme 'ordinary'.
          const entryRecord: Partial<Record<HarvestGrade, number>> =
            typeof entry === 'number' ? { ordinary: entry } : (entry as Partial<Record<HarvestGrade, number>>);
          for (const grade of GRADES_ORDER) {
            const qty = entryRecord[grade] ?? 0;
            if (qty <= 0) continue;
            out.push({
              cropId,
              emoji: def.emoji,
              labelKey: def.labelKey,
              grade,
              qty,
              affinity: getAffinity(companionSpecies, cropId),
            });
          }
        }
        // Tri : préféré d'abord, puis grade desc
        out.sort((a, b) => {
          const r = AFFINITY_RANK[a.affinity] - AFFINITY_RANK[b.affinity];
          if (r !== 0) return r;
          return GRADES_ORDER.indexOf(a.grade) - GRADES_ORDER.indexOf(b.grade);
        });
        return out;
      }, [inventory, companionSpecies]);

      const handlePick = useCallback(
        (row: Row) => {
          Haptics.selectionAsync().catch(() => {});
          onPick(row.cropId, row.grade);
          onClose();
        },
        [onPick, onClose],
      );

      return (
        <Modal
          visible={visible}
          animationType="slide"
          presentationStyle="pageSheet"
          onRequestClose={onClose}
        >
          <SafeAreaView style={[styles.safe, { backgroundColor: colors.bg }]}>
            <ModalHeader title="Nourrir le compagnon" onClose={onClose} />
            {rows.length === 0 ? (
              <View style={styles.empty}>
                <Text style={[styles.emptyText, { color: colors.textMuted }]}>
                  Récoltez quelque chose d'abord 🌱
                </Text>
              </View>
            ) : (
              <ScrollView contentContainerStyle={styles.list}>
                {rows.map((row, idx) => {
                  const isHated = row.affinity === 'hated';
                  const isPreferred = row.affinity === 'preferred';
                  return (
                    <Pressable
                      key={`${row.cropId}-${row.grade}-${idx}`}
                      onPress={() => handlePick(row)}
                      style={({ pressed }) => [
                        styles.row,
                        {
                          backgroundColor: colors.surface,
                          borderColor: isPreferred ? primary : colors.border,
                          opacity: isHated ? 0.55 : pressed ? 0.85 : 1,
                          borderWidth: isPreferred ? 2 : 1,
                        },
                      ]}
                    >
                      <Text style={styles.emoji}>{row.emoji}</Text>
                      <View style={styles.info}>
                        <Text style={[styles.name, { color: colors.text }]}>
                          {t(row.labelKey)}
                          {isPreferred && ' ❤️'}
                          {isHated && ' 😖'}
                        </Text>
                        <Text style={[styles.grade, { color: colors.textMuted }]}>
                          {GRADE_EMOJI[row.grade]} {GRADE_LABEL[row.grade]}
                        </Text>
                      </View>
                      <Text style={[styles.qty, { color: colors.text }]}>x{row.qty}</Text>
                    </Pressable>
                  );
                })}
              </ScrollView>
            )}
          </SafeAreaView>
        </Modal>
      );
    }

    const styles = StyleSheet.create({
      safe: { flex: 1 },
      list: { padding: Spacing.lg, gap: Spacing.sm },
      row: {
        flexDirection: 'row',
        alignItems: 'center',
        padding: Spacing.md,
        borderRadius: Radius.md,
        gap: Spacing.md,
      },
      emoji: { fontSize: 32 },
      info: { flex: 1 },
      name: { fontSize: FontSize.md, fontWeight: FontWeight.semiBold },
      grade: { fontSize: FontSize.sm, marginTop: 2 },
      qty: { fontSize: FontSize.md, fontWeight: FontWeight.bold },
      empty: { flex: 1, justifyContent: 'center', alignItems: 'center', padding: Spacing.xl },
      emptyText: { fontSize: FontSize.md },
    });
    ```

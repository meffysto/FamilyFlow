    ```typescript
    const handleDeliver = useCallback(async (visitor: ActiveVisitor) => {
      if (!activeProfile) return;
      const result = await deliverVisitor(activeProfile.id, visitor.instanceId);
      if (result.ok && result.reward) {
        await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
        showToast(`Livré ! +${result.reward.coins} 🍃${result.reward.loot ? ' + 🎁' : ''}`);
      } else {
        await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Error);
      }
    }, [activeProfile, deliverVisitor, showToast]);

    const handleDismiss = useCallback((visitor: ActiveVisitor) => {
      if (!activeProfile) return;
      Alert.alert('Décliner ce visiteur ?', 'Il repartira sans rien et il faudra attendre avant qu\'il revienne.', [
        { text: 'Annuler', style: 'cancel' },
        { text: 'Décliner', style: 'destructive', onPress: async () => {
          await dismissVisitor(activeProfile.id, visitor.instanceId);
          await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Warning);
        }},
      ]);
    }, [activeProfile, dismissVisitor]);

    const handleForceSpawn = useCallback(async () => {
      if (!activeProfile) return;
      const v = await forceSpawn(activeProfile.id);
      await Haptics.selectionAsync();
      if (!v) showToast('Aucun visiteur éligible (arbre trop jeune ou cap atteint)');
    }, [activeProfile, forceSpawn, showToast]);
    ```

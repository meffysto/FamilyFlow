    export function receivedForProfile(notes: LoveNote[], profileId: string, now = new Date()): LoveNote[] {
      const tier = (n: LoveNote): number => {
        // Tier 1 (top) : révélées + non-lues
        if (isRevealed(n, now) && n.status !== 'read') return 1;

        .filter((n) => n.to === profileId)

          const ta = tier(a);
          const tb = tier(b);
          if (ta !== tb) return ta - tb;
          return b.createdAt.localeCompare(a.createdAt);
        });
    }

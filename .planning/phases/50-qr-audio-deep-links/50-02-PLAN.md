```typescript
// lib/deep-link.ts вЂ” Parsing deep links Family Vault.
// Phase 50 (QR audio) : extraction storyId depuis family-vault://story/<id>.
// Pattern pur testable Jest, zГ©ro dГ©pendance native.

export interface ParsedStoryLink {
  storyId: string;
}

const STORY_SCHEME = 'family-vault:';
const STORY_HOST = 'story';

/**
 * Parse une URL deep link Family Vault story.
 * @param href URL brute reГ§ue (depuis QR scan ou Linking)
 * @returns `{ storyId }` si match, `null` sinon (jamais throw)
 *
 * @example
 * parseStoryDeepLink('family-vault://story/abc-123') // { storyId: 'abc-123' }
 * parseStoryDeepLink('family-vault://story/')        // null
 * parseStoryDeepLink('https://x.com/story/abc')       // null
 */
export function parseStoryDeepLink(href: string): ParsedStoryLink | null {
  if (typeof href !== 'string' || href.length === 0) return null;
  try {
    const url = new URL(href);
    if (url.protocol !== STORY_SCHEME) return null;
    if (url.host !== STORY_HOST) return null;
    // Premier segment du path uniquement, trim slashes
    const firstSegment = url.pathname.replace(/^\/+/, '').split('/')[0] ?? '';
    if (!firstSegment) return null;
    return { storyId: decodeURIComponent(firstSegment) };
  } catch {
    return null;
  }
}
```

```tsx
// Fallback FR + retour bibliothГЁque si id introuvable.

import { useLocalSearchParams, router, Stack } from 'expo-router';

import { useStoryVoice } from '../../contexts/StoryVoiceContext';

import { useThemeColors } from '../../contexts/ThemeContext';
import { FullscreenStoryReader } from '../../components/stories/FullscreenStoryReader';

export default function StoryDeepLinkRoute() {
  const params = useLocalSearchParams<{ id: string }>();
  const id = typeof params.id === 'string' ? params.id : '';
  const { stories, isReady } = useVault();
  const storyVoice = useStoryVoice();
  const { showToast } = useToast();
  const colors = useThemeColors();
  const [hasResolved, setHasResolved] = useState(false);

  const histoire = useMemo(() => {
    if (!isReady || !id) return null;
    return stories.find((s) => s.id === id) ?? null;
  }, [stories, id, isReady]);

  // Gate isReady : on attend l'hydratation du vault avant de chercher l'histoire
  // (cold start scan QR в†’ route monte avant que stories[] soit peuplГ©)
  useEffect(() => {
    if (!isReady) return;
    if (hasResolved) return;
    setHasResolved(true);
    if (!histoire) {
      if (__DEV__) console.warn('[story-deep-link] id introuvable:', id);
      showToast('Histoire introuvable', 'error');
      // Retour bibliothГЁque (les histoires vivent dans l'onglet journal)
      router.replace('/(tabs)/journal');
    }
  }, [isReady, hasResolved, histoire, id, showToast]);

  // Spinner pendant chargement vault OU avant rГ©solution
  if (!isReady || !hasResolved) {
    return (
      <View style={[styles.loadingContainer, { backgroundColor: colors.background }]}>
        <ActivityIndicator color={colors.primary} size="large" />
      </View>
    );
  }

  // Histoire introuvable : toast dГ©jГ  Г©mis + redirection en cours
  if (!histoire) return null;

  return (
    <>
      <Stack.Screen options={{ headerShown: false, presentation: 'fullScreenModal' }} />
      <FullscreenStoryReader
        histoire={histoire}
        voiceConfig={storyVoice.voiceConfig}
        elevenLabsKey={storyVoice.elevenLabsKey}
        fishAudioKey={storyVoice.fishAudioKey}
        autoplay
        onClose={() => router.replace('/(tabs)/journal')}
      />
    </>
  );
}

const styles = StyleSheet.create({
  loadingContainer: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
  },
});
```

       ```tsx
       import React, { useCallback, useEffect, useState } from 'react';
       import { View, Text, Pressable, Alert, Linking, Platform, StyleSheet } from 'react-native';
       import * as Print from 'expo-print';
       import * as Haptics from 'expo-haptics';
       import { useTranslation } from 'react-i18next';
       import { Save, Eye, Printer } from 'lucide-react-native';
       import { useThemeColors } from '../../contexts/ThemeContext';
       import { Spacing, Typography } from '../../constants';
       import { LuluInstructionsModal } from './LuluInstructionsModal';

       interface Props {
         uri: string;
         storyTitle: string;
         onDone: () => void;
       }

       function PostExportViewImpl({ uri, storyTitle, onDone }: Props) {
         const { t } = useTranslation();
         const { colors, primary } = useThemeColors();
         const [luluOpen, setLuluOpen] = useState(false);

         // Haptic Medium au mount (succès génération) — CONTEXT.md
         useEffect(() => {
           Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
         }, []);

         const handleSave = useCallback(async () => {
           Haptics.selectionAsync();
           try {
             // Lazy require pour gérer le cas dev-client pas rebuild — RESEARCH Pitfall 1
             const Sharing = await import('expo-sharing');
             if (await Sharing.isAvailableAsync()) {
               await Sharing.shareAsync(uri, {
                 mimeType: 'application/pdf',
                 UTI: 'com.adobe.pdf',
                 dialogTitle: t('impressions:share.dialogTitle', { title: storyTitle }),
               });
             } else {
               Alert.alert(t('impressions:errors.sharingUnavailable'));
               await Linking.openURL(uri);
             }
           } catch (err) {
             if (__DEV__) console.warn('[PostExportView] save error', err);
             // Fallback gracieux si expo-sharing pas dispo (rebuild dev-client en cours)
             try { await Linking.openURL(uri); } catch { /* silent */ }
           }
         }, [uri, storyTitle, t]);

         const handlePreview = useCallback(async () => {
           Haptics.selectionAsync();
           try {
             if (Platform.OS === 'ios') await Print.printAsync({ uri });
             else await Linking.openURL(uri);
           } catch (err) {
             if (__DEV__) console.warn('[PostExportView] preview error', err);
           }
         }, [uri]);

         const handleLulu = useCallback(() => {
           Haptics.selectionAsync();
           setLuluOpen(true);
         }, []);

         return (
           <View style={[styles.container, { backgroundColor: colors.bg }]}>
             <Text style={[styles.title, { color: colors.text }]}>{t('impressions:postExport.title')}</Text>
             <Text style={[styles.subtitle, { color: colors.muted }]}>{t('impressions:postExport.subtitle')}</Text>

             <ActionButton Icon={Save} title={t('impressions:postExport.save.title')} description={t('impressions:postExport.save.description')} onPress={handleSave} colors={colors} primary={primary} />
             <ActionButton Icon={Eye} title={t('impressions:postExport.preview.title')} description={t('impressions:postExport.preview.description')} onPress={handlePreview} colors={colors} primary={primary} />
             <ActionButton Icon={Printer} title={t('impressions:postExport.lulu.title')} description={t('impressions:postExport.lulu.description')} onPress={handleLulu} colors={colors} primary={primary} />

             <Pressable onPress={onDone} style={[styles.done, { borderColor: colors.border }]}>
               <Text style={[styles.doneLabel, { color: colors.text }]}>{t('impressions:postExport.doneBtn')}</Text>
             </Pressable>

             <LuluInstructionsModal visible={luluOpen} onClose={() => setLuluOpen(false)} />
           </View>
         );
       }

       function ActionButton({ Icon, title, description, onPress, colors, primary }: any) {
         return (
           <Pressable onPress={onPress} style={[styles.action, { backgroundColor: colors.card }]}>
             <Icon size={28} color={primary} />
             <View style={styles.actionText}>
               <Text style={[styles.actionTitle, { color: colors.text }]}>{title}</Text>
               <Text style={[styles.actionDesc, { color: colors.muted }]}>{description}</Text>
             </View>
           </Pressable>
         );
       }

       export const PostExportView = React.memo(PostExportViewImpl);

       const styles = StyleSheet.create({
         container: { flex: 1, padding: Spacing.lg },
         title: { ...Typography.h1, marginBottom: Spacing.xs },
         subtitle: { ...Typography.body, marginBottom: Spacing.xl },
         action: { flexDirection: 'row', alignItems: 'center', padding: Spacing.lg, borderRadius: 12, marginBottom: Spacing.md },
         actionText: { flex: 1, marginLeft: Spacing.md },
         actionTitle: { ...Typography.h3, marginBottom: 2 },
         actionDesc: { ...Typography.caption },
         done: { padding: Spacing.lg, borderRadius: 12, borderWidth: 1, alignItems: 'center', marginTop: Spacing.lg },
         doneLabel: { ...Typography.h3 },
       ```

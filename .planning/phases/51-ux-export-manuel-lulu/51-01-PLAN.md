```ts
type ExportPhase =
  | { kind: 'select' }
  | { kind: 'generating'; step: 'assets' | 'render' | 'hash' | 'print' }
  | { kind: 'ready'; uri: string; perfMs: number; entry: BookManifestEntry }
  | { kind: 'post-export'; uri: string; storyTitle: string };  // branché 51-03

type ExportAction =
  | { type: 'SELECT_STORY'; story: BedtimeStory }
  | { type: 'START_GENERATION' }
  | { type: 'STEP_ADVANCE'; step: ExportPhase extends { step: infer S } ? S : never }
  | { type: 'GENERATION_DONE'; uri: string; perfMs: number; entry: BookManifestEntry }
  | { type: 'GENERATION_ERROR' }
  | { type: 'GO_POST_EXPORT'; uri: string; storyTitle: string }
  | { type: 'RESET' };
```

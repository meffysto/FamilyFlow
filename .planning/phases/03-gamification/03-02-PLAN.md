```typescript
/**
 * quests.ts — Logique pure pour les quetes familiales cooperatives
 */
import type { Defi } from '../types';

/**
 * Calcule la progression cumulee d'une quete cooperative.
 * Compte le nombre total de check-ins completes par TOUS les participants.
 */
export function getCoopProgress(defi: Defi): number {
  return defi.progress.filter(e => e.completed).length;
}

/**
 * Determine si une quete cooperative a atteint son objectif.
 * Retourne false si ce n'est pas une quete cooperative ou si target n'est pas defini.
 */
export function isCoopQuestComplete(defi: Defi): boolean {
  if (defi.questType !== 'cooperative' || !defi.target) return false;
  return getCoopProgress(defi) >= defi.target;
}

/**
 * Calcule le pourcentage de progression d'une quete cooperative (0-100).
 */
export function getCoopProgressPercent(defi: Defi): number {
  if (!defi.target || defi.target <= 0) return 0;
  return Math.min(100, Math.round((getCoopProgress(defi) / defi.target) * 100));
}
```

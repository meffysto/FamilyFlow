export interface FlushQueueDeps {
  profiles: Profile[];
}

/**
 * Drain offline-reason items. Daily-review items restent en queue
 * (consommés par PayoutQueueModal Plan 03, pas par le flush auto).
 */
export async function flushOfflineQueue(deps: FlushQueueDeps): Promise<{ paid: number; remaining: number; failed: number }> {
  if (!await isLightningEnabled()) return { paid: 0, remaining: 0, failed: 0 };
  const config = await loadFamilyConfig();
  if (!config) return { paid: 0, remaining: 0, failed: 0 };
  const queue = await loadQueue();
  const offlineItems = queue.filter(q => q.reason === 'offline');
  let paid = 0, failed = 0;
  for (const item of offlineItems) {
    // re-resolve member depuis config.members
    const wallet = config.members.find(m => m.profileId === item.profileId);
    const profile = deps.profiles.find(p => p.id === item.profileId);
    if (!wallet || !profile) {
      await removeFromQueue(item.taskId, item.profileId);
      await appendAudit({...status:'failed', error:'wallet or profile missing'});
      failed++; continue;
    }
    try {
      await executePayout({
        task: { id: item.taskId, text: '(retry offline)', completed: true, mentions: [] } as any,
        recipient: { profileId: item.profileId, profile, wallet },
        config,
        source: 'flush-offline',
      });
      await removeFromQueue(item.taskId, item.profileId);
      paid++;
    } catch (err: any) {
      const updated = await incrementAttempt(item.taskId, item.profileId, String(err?.message ?? err));
      if (updated && updated.attemptCount >= 5) {
        await removeFromQueue(item.taskId, item.profileId);
        await appendAudit({...status:'failed', error:'max_attempts'});
        failed++;
      }
    }
  }
  const after = await loadQueue();
  return { paid, remaining: after.length, failed };
}

function localTodayISO(): string {
  const d = new Date();
  const pad = (n: number) => String(n).padStart(2, '0');
  return `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`;
}

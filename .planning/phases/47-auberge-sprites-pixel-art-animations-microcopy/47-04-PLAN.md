       const flashOpacity = useSharedValue(0);
       const particleY = useSharedValue(0);
       const particleOpacity = useSharedValue(0);

       useEffect(() => {
         if (deliverTrigger == null || deliverTrigger === 0) return;
         if (reducedMotion) return;

           withSpring(1, SPRING_CONFIG),

       **Approche pragmatique** : déclencher l'incrément `deliveryTriggers` AVANT `await deliverVisitor` (animation = optimiste, démarre instant). Si la livraison échoue, l'animation aura déjà joué — c'est tolérable (rare cas).

       ```ts
       const handleDeliver = useCallback(async (visitor: ActiveVisitor) => {
         if (!activeProfile) return;
         // Trigger animation OPTIMISTE
         setDeliveryTriggers(prev => ({ ...prev, [visitor.instanceId]: (prev[visitor.instanceId] ?? 0) + 1 }));
         try {
           const result = await deliverVisitor(activeProfile.id, visitor.instanceId);
           if (result.ok && result.reward) {
             await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Success);
             const def = getVisitorDef(visitor.visitorId);
             const name = VISITOR_LABELS_FR[visitor.visitorId]?.name ?? def?.id ?? 'Visiteur';
             showToast(
               `${name} repart ravi(e) ! +${result.reward.coins} 🍃${result.reward.loot ? ' + 🎁' : ''}`,
               'success',
             );
           } else {
             await Haptics.notificationAsync(Haptics.NotificationFeedbackType.Error);
           }
         } catch (e) {
           if (__DEV__) console.warn('[AubergeSheet] deliver error', e);
         }
       }, [activeProfile, deliverVisitor, showToast]);
       ```

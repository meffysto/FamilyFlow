**useEffect montage** — charger les 3 sources async en parallele :
```typescript
useEffect(() => {
  let mounted = true;
  Promise.all([
    isSemanticCouplingEnabled(),
    loadOverrides(),
    loadWeekStats(),
  ]).then(([enabled, ovr, stats]) => {
    if (!mounted) return;
    setMasterEnabled(enabled);
    setOverrides(ovr);
    setWeekStats(stats);
    setLoading(false);
  });
  return () => { mounted = false; };
}, []);
```

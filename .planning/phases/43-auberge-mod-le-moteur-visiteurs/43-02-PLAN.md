    ```typescript
    import { parseFarmProfile, serializeFarmProfile } from '../parser';
    import type { FarmProfileData } from '../types';

    const baseProfile = (): FarmProfileData => ({
      // remplir les champs minimums obligatoires de FarmProfileData
      // (regarder lib/types.ts pour les non-optionnels — coins, level, etc.)
      // Si tous les champs sont optionnels, partir de {}.
    } as FarmProfileData);

    describe('Phase 43 — Persistance Auberge dans farm-{id}.md', () => {
      describe('Backward compat', () => {
        it('parse un profil sans champs auberge_* sans erreur', () => {
          const md = '---\nname: Lucas\n---\n# Farm\n';
          const data = parseFarmProfile(md);
          expect(data.auberge_visitors).toBeUndefined();
          expect(data.auberge_reputations).toBeUndefined();
          expect(data.auberge_last_spawn).toBeUndefined();
          expect(data.auberge_total_deliveries).toBeUndefined();
        });
      });

      describe('Round-trip', () => {
        it('round-trip lossless avec champs auberge_*', () => {
          const data: FarmProfileData = {
            ...baseProfile(),
            auberge_visitors: 'vis_1234_abcd:hugo_boulanger:2026-04-29T10:00:00Z',
            auberge_reputations: 'hugo_boulanger:2:1:0:2026-04-29T10:00:00Z',
            auberge_last_spawn: '2026-04-29T10:00:00.000Z',
            auberge_total_deliveries: 7,
          };
          const md = serializeFarmProfile('Lucas', data);
          const parsed = parseFarmProfile(md);
          expect(parsed.auberge_visitors).toBe(data.auberge_visitors);
          expect(parsed.auberge_reputations).toBe(data.auberge_reputations);
          expect(parsed.auberge_last_spawn).toBe(data.auberge_last_spawn);
          expect(parsed.auberge_total_deliveries).toBe(7);
        });
      });

      describe('Sérialisation conditionnelle', () => {
        it('skip auberge_total_deliveries si === 0', () => {
          const data: FarmProfileData = {
            ...baseProfile(),
            auberge_total_deliveries: 0,
          };
          const md = serializeFarmProfile('Lucas', data);
          expect(md).not.toContain('auberge_total_deliveries');
        });
        it('skip auberge_visitors si chaîne vide', () => {
          const data: FarmProfileData = {
            ...baseProfile(),
            auberge_visitors: '',
          };
          const md = serializeFarmProfile('Lucas', data);
          expect(md).not.toContain('auberge_visitors:');
        });
      });
    });
    ```

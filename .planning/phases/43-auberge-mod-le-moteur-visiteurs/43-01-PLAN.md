```typescript
// Phase 43 — Auberge
export type VisitorRarity = 'common' | 'uncommon' | 'rare';
export type VisitorRequestSource = 'building' | 'crop' | 'crafted';
export type VisitorStatus = 'active' | 'delivered' | 'expired';

export interface VisitorRequestItem {
  itemId: string;
  source: VisitorRequestSource;
  quantity: number; // snapshot figé (résolu au spawn depuis [min,max])
}

export interface ActiveVisitor {
  visitorId: string;       // ref VisitorDefinition.id
  instanceId: string;      // vis_{ts}_{rand}
  arrivedAt: string;       // ISO
  deadlineAt: string;      // ISO
  request: VisitorRequestItem[];
  status: VisitorStatus;
  rewardCoins: number;     // snapshot à la création
  rewardLootKey?: string;  // résolu à la livraison uniquement
}

export interface VisitorReputation {
  visitorId: string;
  level: number;           // 0-5
  successCount: number;
  failureCount: number;
  lastSeenAt: string;      // ISO
}

/**
 * État Auberge persisté par profil. Le tableau `visitors` ne contient que
 * les visiteurs récents (actifs + delivered/expired < 7 jours). Le compteur
 * `totalDeliveries` agrège tout l'historique livré et n'est JAMAIS dérivé
 * du tableau (Pitfall 8 — éviter double-count après archivage).
 */
export interface AubergeState {
  visitors: ActiveVisitor[];
  reputations: VisitorReputation[];
  lastSpawnAt?: string;    // ISO
  totalDeliveries: number;
}
```

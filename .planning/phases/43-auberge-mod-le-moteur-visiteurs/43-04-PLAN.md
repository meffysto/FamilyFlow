    // hooks/useAuberge.ts — Phase 43
    // Hook orchestrateur Auberge. Pattern useExpeditions autonome (pas de modif VaultContext).

    import { useCallback, useMemo } from 'react';
    import { useVault } from '../contexts/VaultContext';
    import { parseFarmProfile, serializeFarmProfile } from '../lib/parser';
    import {
      parseAuberge, serializeAuberge,
      deliverVisitor as engineDeliver,
      dismissVisitor as engineDismiss,
      expireVisitors as engineExpire,
      spawnVisitor as engineSpawn,
      canDeliver as engineCanDeliver,
      getActiveVisitors,
      getTotalReputation,
    } from '../lib/mascot/auberge-engine';
    import type {
      AubergeState, ActiveVisitor, VisitorRequestItem, VisitorReputation,
    } from '../lib/mascot/types';

    const farmFile = (profileId: string) => `farm-${profileId}.md`;

    export function useAuberge() {
      const { vault, profiles, currentProfile } = useVault();
      // Adapter selon l'API réelle de useVault — peut-être profile, activeProfile, etc.

      // Lecture dérivée
      const state: AubergeState = useMemo(() => parseAuberge({
        visitors: currentProfile?.auberge_visitors,
        reputations: currentProfile?.auberge_reputations,
        lastSpawn: currentProfile?.auberge_last_spawn,
        totalDeliveries: currentProfile?.auberge_total_deliveries,
      }), [
        currentProfile?.auberge_visitors,
        currentProfile?.auberge_reputations,
        currentProfile?.auberge_last_spawn,
        currentProfile?.auberge_total_deliveries,
      ]);

      const visitors = state.visitors;
      const activeVisitors = useMemo(() => getActiveVisitors(state), [state]);
      const reputations = state.reputations;
      const totalDeliveries = state.totalDeliveries;
      const totalReputation = useMemo(() => getTotalReputation(state), [state]);

      // Helper : write atomique
      const writeAuberge = useCallback(async (
        profileId: string,
        mutator: (state: AubergeState, farmData: any) => { state: AubergeState; sideEffects?: () => Promise<void> }
      ) => {
        if (!vault) return;
        const file = farmFile(profileId);
        const content = await vault.readFile(file).catch(() => '');
        const farmData = parseFarmProfile(content);
        const currentState = parseAuberge({
          visitors: farmData.auberge_visitors,
          reputations: farmData.auberge_reputations,
          lastSpawn: farmData.auberge_last_spawn,
          totalDeliveries: farmData.auberge_total_deliveries,
        });
        const { state: nextState, sideEffects } = mutator(currentState, farmData);
        const serialized = serializeAuberge(nextState);
        farmData.auberge_visitors = serialized.visitors || undefined;
        farmData.auberge_reputations = serialized.reputations || undefined;
        farmData.auberge_last_spawn = serialized.lastSpawn;
        farmData.auberge_total_deliveries = serialized.totalDeliveries > 0 ? serialized.totalDeliveries : undefined;
        if (sideEffects) await sideEffects();
        const profileName = profiles.find((p: any) => p.id === profileId)?.name ?? profileId;
        await vault.writeFile(file, serializeFarmProfile(profileName, farmData));
      }, [vault, profiles]);

      // Setters
      const deliverVisitor = useCallback(async (profileId: string, instanceId: string) => {
        // ... lire farmData, extraire inventaires (FarmInventory, HarvestInventory, CraftedItem[]),
        //     vérifier canDeliver, appliquer déduction sur farmData, créditer coins, écrire.
        // Retourner { ok, reward?, missing? }.
      }, [writeAuberge]);

      const dismissVisitor = useCallback(async (profileId: string, instanceId: string) => {
        await writeAuberge(profileId, (s) => ({
          state: engineDismiss(s, instanceId, new Date()).state,
        }));
      }, [writeAuberge]);

      const tickAuberge = useCallback(async (profileId: string) => {
        let spawned: ActiveVisitor | undefined;
        let expired: ActiveVisitor[] = [];
        await writeAuberge(profileId, (s, farmData) => {
          const now = new Date();
          const expireResult = engineExpire(s, now);
          expired = expireResult.expired;
          let next = expireResult.state;
          // Lire treeStage depuis farmData
          const treeStage = farmData.tree_stage ?? 'pousse'; // adapter au champ réel
          const totalRep = getTotalReputation(next);
          const spawnResult = engineSpawn(next, treeStage, now, totalRep);
          if (spawnResult) {
            next = spawnResult.state;
            spawned = spawnResult.visitor;
          }
          return { state: next };
        });
        return { spawned, expired };
      }, [writeAuberge]);

      return {
        visitors, activeVisitors, reputations, totalDeliveries, totalReputation,
        deliverVisitor, dismissVisitor, tickAuberge,
      };
    }

```typescript
// Composant inline dans Budget.tsx
function ReceiptDropZone({ onFile }: { onFile: (file: File) => void }) {
  const [isDragOver, setIsDragOver] = useState(false);
  return (
    <div
      className={`receipt-drop-zone${isDragOver ? ' drag-over' : ''}`}
      onDragOver={(e) => { e.preventDefault(); setIsDragOver(true); }}
      onDragLeave={() => setIsDragOver(false)}
      onDrop={(e) => {
        e.preventDefault();
        setIsDragOver(false);
        const file = e.dataTransfer.files[0];
        if (file && file.type.startsWith('image/')) onFile(file);
      }}
    >
      <span className="drop-icon">📸</span>
      <p>Deposer une photo de recu ici</p>
      <p className="drop-subtitle">ou</p>
      <label className="upload-btn">
        Choisir un fichier
        <input
          type="file"
          accept="image/*"
          style={{ display: 'none' }}
          onChange={(e) => { const f = e.target.files?.[0]; if (f) onFile(f); }}
        />
      </label>
    </div>
  );
}
```

**2. Conversion image -> base64 :**
```typescript
async function fileToBase64(file: File): Promise<string> {
  return new Promise((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = () => resolve((reader.result as string).split(',')[1]); // strip data:image/...;base64, prefix
    reader.onerror = reject;
    reader.readAsDataURL(file);
  });
}
```

```typescript
import type { Profile } from '../types';

/** Type-token ratio : tokens uniques / total. 0 sur entrée vide. */
export function typeTokenRatio(text: string): number {
  const tokens = text.toLowerCase().split(/\s+/).filter((t) => /[a-zà-ÿ]/i.test(t));
  if (tokens.length === 0) return 0;
  const unique = new Set(tokens);
  return unique.size / tokens.length;
}

/** Compte occurrences d'un pattern (must be /g/). */
export function countOccurrences(text: string, pattern: RegExp): number {
  if (!pattern.global) throw new Error('countOccurrences requires global regex');
  return (text.match(pattern) ?? []).length;
}

/**
 * Overlap n-grams (mots) entre deux textes après normalisation FR.
 * Retourne ratio [0..1] = |intersection| / |union des n-grams de A|.
 * Normalise : lowercase, strip prénoms enfants connus (sinon prénom commun fausse le calcul).
 */
export function ngramOverlap(a: string, b: string, n: number, knownNames: string[] = []): number {
  const norm = (s: string) => {
    let out = s.toLowerCase();
    for (const name of knownNames) {
      out = out.replace(new RegExp(`\\b${name.toLowerCase()}\\b`, 'g'), 'enfant');
    }
    return out.split(/\s+/).filter(Boolean);
  };
  const grams = (tokens: string[]) => {
    const set = new Set<string>();
    for (let i = 0; i <= tokens.length - n; i++) set.add(tokens.slice(i, i + n).join(' '));
    return set;
  };
  const A = grams(norm(a));
  const B = grams(norm(b));
  if (A.size === 0) return 0;
  let common = 0;
  for (const g of A) if (B.has(g)) common++;
  return common / A.size;
}

/** Anonymise le texte avant envoi au LLM-judge — réutilise pattern lib/anonymizer.ts. */
export function anonymizeStoryText(text: string, child: Profile): string {
  const name = child.prenom ?? child.nom ?? '';
  if (!name) return text;
  return text.replace(new RegExp(`\\b${name}\\b`, 'gi'), 'Enfant');
}
```
